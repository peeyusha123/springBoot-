SELECT *FROM ERP_PURCHASE_ORDER_PROJECT POP;

		SELECT CONCAT(POH.PO_ID, '') AS PO_ID,
		MAX(PEL.VENDOR_NAME) VENDOR_NAME,
		MAX(PEL.MATERIAL_ITEM_ID) ITEM_ID,
		-- MAX(PS.Z_ITM_COMP_CD) ITEM_COMPONENT,
		MAX(PEL.MATERIAL_DESC) ITEM_DESCRIPTION,
		MAX(PEL.QTY) PO_QTY,
		MAX(POP.PO_DUE_DATE) DUE_DATE,
		-- POP.BUYER,
		CONCAT(CONCAT(MAX(SU.LAST_NAME), ', '), MAX(SU.FIRST_NAME)) as BUYER,
		MAX(POH.STATUS) AS STATUS,
		MAX(ER.VALUE) AS STATUS_DESCRIPTION,
		MAX(POH.PO_REFERENCE) DESCRIPTION,
		MAX(POP.SITE_PROJECTS_ID) SITE_PROJECTS_ID,
		MAX(POP.WBS_ELEMENT) PEOPLESOFT_PROJECT_NUMBER,
		MAX(POP.WBS_ELEMENT_EXP) PEOPLESOFT_PROJECT_NUMBER_EXP,
		MAX(POH.ERP_CREATED_DATE) ORDER_DATE,
		-- TO_DATE(PTD.RECEIVED_DATE,'MM/DD/YYYY') as RECEIVED_DATE,
		TO_DATE(TO_CHAR(TO_TIMESTAMP(MAX(GR.LAST_MODIFIED_DT)),'MM/DD/YYYY'),'MM/DD/YYYY') AS RECEIVED_DATE ,
		MAX(PEL.ITEM_GROUP) AS ITEM_GROUP,
		MAX(GR.GR_QTY) AS RECEIVED_QTY,
		PEL.LINE_NUMBER,
		PEL.SCHED_NBR,
		COUNT(PEL.PO_NUMBER) NUMBER_PO_LINES, 
		SUM(PEL.CONTRACTPRICE * PEL.QTY) PO_TOTAL_AMOUNT,
		 'po2/request/view/' || POH.PURCHASE_ORDER_REQUEST_ID ||'/cart' AS SINGLE_PO_LINK,
		MAX(POH.S4_PURCHASE_DOC_NUM) AS S4_PURCHASE_DOC_NUM
	FROM ERP_PURCHASE_ORDER_PROJECT POP
	INNER JOIN ERP_PURCHASE_ORDER_HEADER POH ON POP.PURCHASE_ORDER_REQUEST_ID = POH.PURCHASE_ORDER_REQUEST_ID
	INNER JOIN ERP_PO_EQUIPMENT_LINEITEM PEL ON --POH.PO_ID = CAST(PEL.PO_NUMBER AS NUMBER) AND 
    PEL.PURCHASE_ORDER_PROJECT_ID = POP.PURCHASE_ORDER_PROJECT_ID  AND POH.PO_SEQUENCE = PEL.PO_SEQUENCE
	LEFT JOIN MM_MATERIAL_ITEM PS ON PS.MAT_NUM = PEL.MATERIAL_ITEM_ID -- AND PS.SETID = 'NETWK'
--	LEFT JOIN ERP_PO_TRACKER PTD ON POP.PURCHASE_ORDER_REQUEST_ID = PTD.PURCHASE_ORDER_REQUEST_ID
	LEFT JOIN ERP_PO_LINE_ITEM_STATUS GR on GR.PO_ID = PEL.PO_NUMBER and GR.ERP_PO_EQUIPMENT_LINE_ITEM_ID = PEL.LINE_NUMBER
	-- LEFT JOIN PROJ_PSLC_PROJECT_INFO PPI ON PPI.SITE_PROJECTS_ID = POP.SITE_PROJECTS_ID and PPI.PSLC_PROJECTS_ID = POP.PEOPLESOFT_PROJECT_NUMBER
	LEFT OUTER JOIN SEC_USERS SU on LOWER(POH.BUYER_ID) = LOWER(SU.USER_ID)
	LEFT JOIN ENUM_REFERENCE ER ON ER.GROUP_NAME = 'STATUS' AND ER.SUB_TYPE='PO_STATUS' AND ER.NAME = POH.STATUS
	--WHERE 1 = 1
	WHERE POP.SITE_PROJECTS_ID = :!SITE_PROJECTS_ID: AND LENGTH(POH.PO_ID) > 0
	--AND ROWNUM <2000
	GROUP BY POH.PURCHASE_ORDER_REQUEST_ID,POH.PO_ID,POH.PO_SEQUENCE,PEL.LINE_NUMBER,PEL.SCHED_NBR
	ORDER BY POH.PO_ID,PEL.LINE_NUMBER,PEL.SCHED_NBR
